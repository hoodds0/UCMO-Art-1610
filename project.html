def load_spritesheet(self, dir):
    """Load Sprite Sheets
            
    Arguments:
        dir -- Folder where Sprite Sheet are
            
    Returns:
        A Dict with a list of (X, Y) Pairs of 10x10 [x][y] for an Image
    """
    path = join("ASSETS", dir, "Sheets")
    images = [f for f in listdir(path) if isfile(join(path, f))]
    all_sheets = {}
    for image in images:
        sheet = pygame.image.load(join(path, image)).convert_alpha()
        spritesX = []
        width = sheet.get_width() // 10
        height = sheet.get_height() // 10
        for x in range(sheet.get_width() // width):
            spritesY = []
            for y in range(sheet.get_height() // height):
                surface = pygame.Surface((width, height), pygame.SRCALPHA)
                rect = pygame.Rect(x * width, y * height, width, height)
                surface.blit(sheet, (0, 0), rect)
                spritesY.append(surface)
                spritesX.append(spritesY)
        all_sheets[image.replace(".png", "")] = spritesX
    return all_sheets
